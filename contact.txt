    <!-- logo轮播JS（平滑滑动） -->
    <script>
    document.addEventListener('DOMContentLoaded', function() {
      const logoRow = document.querySelector('.clients-logos');
      if (!logoRow) return;
      const logoNodes = Array.from(logoRow.querySelectorAll('img'));
      // 包裹每个logo
      if (!logoRow.querySelector('.client-logo-box')) {
        logoRow.innerHTML = '';
        logoNodes.forEach(img => {
          const box = document.createElement('div');
          box.className = 'client-logo-box';
          box.appendChild(img);
          logoRow.appendChild(box);
        });
      }
      const boxes = Array.from(logoRow.querySelectorAll('.client-logo-box'));
      let visibleCount = 7;
      let startIdx = 0;
      let intervalId = null;
      let animating = false;

      function updateVisibleCount() {
        if (window.innerWidth < 600) return 2;
        if (window.innerWidth < 900) return 4;
        return 7;
      }

      function renderLogos() {
        // 只渲染可见数量+1（用于平滑滑动）
        logoRow.innerHTML = '';
        for (let i = 0; i < visibleCount + 1; i++) {
          const idx = (startIdx + i) % boxes.length;
          logoRow.appendChild(boxes[idx].cloneNode(true));
        }
        logoRow.style.transition = 'none';
        logoRow.style.transform = 'translateX(0)';
      }

      function slideNext() {
        if (animating) return;
        animating = true;
        logoRow.style.transition = 'transform 0.8s cubic-bezier(.4,0,.2,1)';
        logoRow.style.transform = `translateX(-${logoRow.firstChild.offsetWidth}px)`;
        setTimeout(() => {
          startIdx = (startIdx + 1) % boxes.length;
          renderLogos();
          animating = false;
        }, 800);
      }

      function startCarousel() {
        if (intervalId) return;
        intervalId = setInterval(slideNext, 3000);
      }
      function stopCarousel() {
        if (intervalId) clearInterval(intervalId);
        intervalId = null;
      }

      // IntersectionObserver，滚动到区块时启动轮播
      const observer = new IntersectionObserver((entries) => {
        entries.forEach(entry => {
          if (entry.isIntersecting) {
            visibleCount = updateVisibleCount();
            renderLogos();
            startCarousel();
          } else {
            stopCarousel();
          }
        });
      }, { threshold: 0.3 });
      observer.observe(logoRow);

      window.addEventListener('resize', () => {
        visibleCount = updateVisibleCount();
        renderLogos();
      });
    });
    </script>
